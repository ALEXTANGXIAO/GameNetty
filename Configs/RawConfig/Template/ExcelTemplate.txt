        {
            if (_configs.ContainsKey(id))
            {
                return _configs[id];
            }
    
            if (check)
            {
                throw new Exception($"(ConfigName) not find {id} Id");
            }
            
            return null;
        }
        public bool TryGet(uint id, out (ConfigName) config)
        {
            config = null;
            
            if (!_configs.ContainsKey(id))
            {
                return false;
            }
                
            config = _configs[id];
            return true;
        }

            {
                (ConfigName) config = List[i];
                _configs.Add(config.Id, config);

        }
